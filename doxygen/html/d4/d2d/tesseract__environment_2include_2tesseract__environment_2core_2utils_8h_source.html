55 inline void getActiveLinkNamesRecursive(std::vector<std::string>& active_links,
56  const tesseract_scene_graph::SceneGraph::ConstPtr& scene_graph,
57  const std::string& current_link,
58  bool active)
59 {
60  // recursively get all active links
61  assert(scene_graph->isTree());
62  if (active)
63  {
64  active_links.push_back(current_link);
65  for (const auto& child_link : scene_graph->getAdjacentLinkNames(current_link))
66  getActiveLinkNamesRecursive(active_links, scene_graph, child_link, active);
67  }
68  else
69  {
70  for (const auto& child_link : scene_graph->getAdjacentLinkNames(current_link))
71  if (scene_graph->getInboundJoints(child_link)[0]->type != tesseract_scene_graph::JointType::FIXED)
72  getActiveLinkNamesRecursive(active_links, scene_graph, child_link, true);
73  else
74  getActiveLinkNamesRecursive(active_links, scene_graph, child_link, active);
75  }
76 }
